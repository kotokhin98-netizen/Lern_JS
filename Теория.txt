// Функция БЕЗ параметров - запрашивает ввод
function getInput() {
    let a = +prompt("Введите число от 0 до 3:");
    checkNumber(a);  // вызываем вторую функцию
}

// Функция С параметром - обрабатывает число
function checkNumber(num) {
    switch (num) {
        case 0:
            alert("Вы ввели 0");
            break;
        case 1:
            alert("Вы ввели 1");
            break;
        case 2:
        case 3:
            alert("Вы ввели 2 и 3");
            break;
        default:
            alert("Число не входит в диапазон");
            break;
    }
}

// Запускаем программу
getInput();


getInput() - функция без параметров, которая:
Запрашивает число через prompt
Преобразует строку в число (+)
Вызывает checkNumber(a) и передает туда число
checkNumber(num) - функция с параметром num, которая:
Проверяет число через switch-case
Для 0 и 1 выводит отдельные сообщения
Для 2 и 3 (без break между ними) выводит одно сообщение
default обрабатывает все остальные числа
Чтобы запустить программу, просто вызовите getInput().

        const hoverBox = document.getElementById('hover-target');
        
        // Слушаем событие "наведение мыши"
        hoverBox.addEventListener('mouseenter', function() {
            this.classList.add('highlight');
        });

        // Слушаем событие "уход мыши"
        hoverBox.addEventListener('mouseleave', function() {
            this.classList.remove('highlight');
        });

        // --- ФУНКЦИЯ ДЛЯ ЗАДАНИЯ 3 ---
        function toggleVisibility() {
            const block = document.getElementById('hidden-block');
            
            // Проверяем текущее значение стиля display
            if (block.style.display === 'none') {
                block.style.display = 'block'; // Показываем
            } else {
                block.style.display = 'none'; // Скрываем
            }
        }

        // --- ФУНКЦИЯ ДЛЯ ЗАДАНИЯ 4 ---
        function changeStyle(color, fontSize) {
            const target = document.getElementById('style-target');
            // Прямое изменение CSS свойств через JS
            target.style.backgroundColor = color;
            target.style.fontSize = fontSize;
        }
